from gluonts.model import Trainer
from gluonts.model.deepar import DEEPAR
from gluonts.trainer import Trainer

# assume we have a pandas dataframe df with a column 'value'
df = pd.read_csv('data.csv')

# define the frequency of the time series
freq = 'H'

# define the prediction length
prediction_length = 10

# define the trainer
trainer = Trainer(
    epochs=10,
    learning_rate=0.01,
    batch_size=32,
    num_workers=4
)

# define the model
model = DEEPAR(
    freq=freq,
    prediction_length=prediction_length,
    trainer=trainer
)

# fit the model
model.fit(df)

# make predictions
forecast = model.predict(df)

# print the forecast
print(forecast)
